Vista previa

{$CLEO}
thread 'EJEMPLO'
while true 
   wait 0
   if or
      Game.isPressedVK(0, Key_Vertical)
      Game.isPressed(Key_R)
   then
      wait 0
      Actor.setWalkStyle($PLAYER_ACTOR, "parkourman")
   else
      wait 1000
      Actor.setWalkStyle($PLAYER_CHAR, "man")
   end
end
