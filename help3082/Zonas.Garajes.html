0299: activate_garage 'BODLAWN'
//Tambien puedes usar la clase Garage de esta forma;
Garage.Activate("GarageID")
//Y puedes usar este opcode para detectar si esta abierto
03B0:   garage 'BODLAWN' door_open
Y bloquearlo de nuevo con: 
02B9: deactivate_garage 'BODLAWN'
